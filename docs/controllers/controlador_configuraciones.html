 26    """

 29    Permite al usuario administrar:

 66

 76    def seleccionar_carpeta_compartida(self):
 77        """Abre un diálogo para seleccionar el directorio de almacenamiento de archivos.
 78
 79        Actualiza el campo visual con la ruta seleccionada por el usuario.
 80        Se normaliza la ruta para compatibilidad con el sistema operativo.
 81        """
 82        carpeta = QFileDialog.getExistingDirectory(self, "Seleccionar Carpeta para Certificados Firmados")
 83        if carpeta:
 84            # Reemplazar slash por backslash para mejor lectura en Windows (opcional)
 85            carpeta = os.path.normpath(carpeta)
 86            self.lineEdit_CarpetaCompartida.setText(carpeta)
 87
 88    def cargar_datos_bd(self):
 89        """Carga la configuración actual desde las variables de entorno a la UI.
 90
 91        Lee los valores de conexión a base de datos (host, puerto, usuario, etc.)
 92        y la ruta de la carpeta compartida, rellenando los campos correspondientes.
 93        """
 94        if config.DB_TYPE == "sqlite":
 95            self.radioBtn_Local.setChecked(True)
 96        else:
 97            self.radioBtn_Online.setChecked(True)
 98
 99        self.lineEdit_Url.setText(os.getenv("DB_HOST", ""))
100        self.lineEdit_Puerto.setText(os.getenv("DB_PORT", "5432"))
101        self.lineEdit_NombreBD.setText(os.getenv("DB_NAME_REMOTE", "academia_prod"))
102        self.lineEdit_Usuario.setText(os.getenv("DB_USER", ""))
103        self.lineEdit_Password.setText(os.getenv("DB_PASS", ""))
104
105        # --- NUEVO: Cargar ruta compartida ---
106        ruta_guardada = os.getenv("SHARED_FOLDER_PATH", config.DEFAULT_SHARED_DIR)
107        self.lineEdit_CarpetaCompartida.setText(ruta_guardada)

109    def al_cambiar_tipo_bd(self):
110        """Ajusta valores por defecto al cambiar entre modo Local y Online.
111
112        Si se selecciona 'Online' y los campos están vacíos, sugiere 'localhost'
113        y el puerto '5432'.
114        """
115        if not self.radioBtn_Local.isChecked():
116            if not self.lineEdit_Url.text(): self.lineEdit_Url.setText("localhost")
117            if not self.lineEdit_Puerto.text(): self.lineEdit_Puerto.setText("5432")

119    def probar_conexion_bd(self):
120        """Realiza una prueba de conexión con los parámetros ingresados.
121
122        - Para SQLite (Local): Verifica la existencia del archivo (simulado).
123        - Para PostgreSQL (Online): Intenta crear un engine y conectar realmente.
124
125        Muestra un mensaje modal con el resultado (Éxito o Error).
126        """
127        if self.radioBtn_Local.isChecked():
128            QMessageBox.information(self, "Prueba", "Conexión local (SQLite) es un archivo, siempre funciona.")
129            return
130
131        host = self.lineEdit_Url.text().strip()
132        port = self.lineEdit_Puerto.text().strip()
133        db_name = self.lineEdit_NombreBD.text().strip()
134        user = self.lineEdit_Usuario.text().strip()
135        password = self.lineEdit_Password.text()
136
137        if not all([host, port, db_name, user, password]):
138            QMessageBox.warning(self, "Campos Vacíos", "Llena todos los campos.")
139            return
140
141        url_prueba = f"postgresql://{user}:{password}@{host}:{port}/{db_name}"
142        QApplication.setOverrideCursor(Qt.CursorShape.WaitCursor)
143
144        try:
145            motor_prueba = create_engine(url_prueba, connect_args={'connect_timeout': 5})
146            with motor_prueba.connect() as conexion:
147                QApplication.restoreOverrideCursor()
148                QMessageBox.information(self, "¡Éxito!", "Conexión exitosa a la base de datos PostgreSQL.")
149        except Exception as e:
150            QApplication.restoreOverrideCursor()
151            QMessageBox.critical(self, "Error", f"Fallo en conexión:\n{str(e)[:200]}...")
152
153    def guardar_configuracion_bd(self):
154        """Persiste la configuración actual en el archivo de entorno (.env).
155
156        Guarda:
157        - Tipo de base de datos (sqlite/postgresql).
158        - Credenciales de conexión (si aplica).
159        - Ruta de la carpeta compartida.
160
161        Notifica al usuario que debe reiniciar para aplicar cambios.
162        """
163        tipo_bd = "sqlite" if self.radioBtn_Local.isChecked() else "postgresql"
164        try:
165            config.actualizar_env("DB_TYPE", tipo_bd)
166
167            if tipo_bd == "postgresql":
168                config.actualizar_env("DB_HOST", self.lineEdit_Url.text().strip())
169                config.actualizar_env("DB_PORT", self.lineEdit_Puerto.text().strip())
170                config.actualizar_env("DB_NAME_REMOTE", self.lineEdit_NombreBD.text().strip())
171                config.actualizar_env("DB_USER", self.lineEdit_Usuario.text().strip())
172                config.actualizar_env("DB_PASS", self.lineEdit_Password.text())
173
174            # --- NUEVO: Guardar la ruta compartida ---
175            ruta_compartida = self.lineEdit_CarpetaCompartida.text().strip()
176            if ruta_compartida:
177                config.actualizar_env("SHARED_FOLDER_PATH", ruta_compartida)
178
179            QMessageBox.information(self, "Éxito",
180                                    "Configuración guardada en .env.\nReinicie la aplicación para aplicar los cambios.")
181        except Exception as e:
182            QMessageBox.critical(self, "Error", f"No se pudo guardar: {e}")

185    def cargar_tipos_certificados(self):
186        """Carga y lista los tipos de certificados disponibles en la base de datos."""
187        self.listWidget_Certificados.clear()
188        try:
189            certificados = self.tipo_cert_model.get_all()
190            for cert in certificados:
191                item = QListWidgetItem(cert.nombre)
192                item.setData(Qt.ItemDataRole.UserRole, cert.id)
193                self.listWidget_Certificados.addItem(item)
194        except Exception as e:
195            print(f"Error cargando certificados: {e}")
196
197    def agregar_certificado(self):
198        """Crea un nuevo tipo de certificado en la base de datos.
199
200        Valida que el nombre no esté vacío y maneja duplicados.
201        """
202        nombre = self.lineEdit_NuevoCertificado.text().strip().upper()
203        if not nombre:
204            QMessageBox.warning(self, "Atención", "Ingrese un nombre para el certificado.")
205            return
206        try:
207            self.tipo_cert_model.create({"nombre": nombre})
208            self.lineEdit_NuevoCertificado.clear()
209            self.cargar_tipos_certificados()
210            QMessageBox.information(self, "Éxito", "Tipo de certificado agregado.")
211        except IntegrityError:
212            QMessageBox.warning(self, "Error", "Este tipo de certificado ya existe.")
213        except Exception as e:
214            QMessageBox.critical(self, "Error", f"No se pudo agregar: {e}")
215
216    def eliminar_certificado(self):
217        """Elimina el tipo de certificado seleccionado tras confirmación."""
218        item_seleccionado = self.listWidget_Certificados.currentItem()
219        if not item_seleccionado: return
220
221        cert_id = item_seleccionado.data(Qt.ItemDataRole.UserRole)
222        cert_nombre = item_seleccionado.text()
223
224        confirm = QMessageBox.question(self, "Confirmar", f"¿Eliminar '{cert_nombre}'?",
225                                       QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No)
226
227        if confirm == QMessageBox.StandardButton.Yes:
228            try:
229                if self.tipo_cert_model.delete(cert_id):
230                    self.cargar_tipos_certificados()
231                    QMessageBox.information(self, "Éxito", "Certificado eliminado.")
232            except Exception as e:
233                QMessageBox.critical(self, "Error", f"No se pudo eliminar: {e}")
234
235    def cargar_instituciones(self):
236        """Carga la lista de instituciones articuladas únicas registradas.
237
238        Filtra valores nulos o vacíos para mostrar solo instituciones válidas.
239        """
240        self.listWidget_Instituciones.clear()
241        try:
242            with SessionLocal() as session:
243                instituciones = session.query(Persona.institucion_articulada).filter(
244                    Persona.institucion_articulada.isnot(None)).filter(
245                    Persona.institucion_articulada != "").distinct().all()
246                for inst in instituciones:
247                    if inst[0]: self.listWidget_Instituciones.addItem(inst[0].upper())
248        except Exception as e:
249            print(f"Error: {e}")
250
251    def al_seleccionar_institucion(self, item):
252        """Coloca el nombre de la institución seleccionada en el campo de edición.
253
254        Args:
255            item (QListWidgetItem): Ítem seleccionado en la lista.
256        """
257        self.lineEdit_RenombrarInstitucion.setText(item.text())

259    def renombrar_institucion(self):
260        """Renombra masivamente una institución en todos los registros de personas.
261
262        Útil para corregir errores tipográficos. Solicita confirmación antes de
263        ejecutar la actualización masiva (UPDATE) en la base de datos.
264        """
265        item_seleccionado = self.listWidget_Instituciones.currentItem()
266        if not item_seleccionado: return
267
268        nombre_antiguo = item_seleccionado.text()
269        nombre_nuevo = self.lineEdit_RenombrarInstitucion.text().strip().upper()
270
271        if not nombre_nuevo or nombre_nuevo == nombre_antiguo: return
272
273        confirm = QMessageBox.question(self, "Confirmar Renombrado",
274                                       f"¿Renombrar '{nombre_antiguo}' a '{nombre_nuevo}' en TODOS los estudiantes?",
275                                       QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No)
276        if confirm == QMessageBox.StandardButton.Yes:
277            try:
278                with SessionLocal() as session:
279                    session.query(Persona).filter(Persona.institucion_articulada == nombre_antiguo).update(
280                        {"institucion_articulada": nombre_nuevo})
281                    session.commit()
282                self.lineEdit_RenombrarInstitucion.clear()
283                self.cargar_instituciones()
284                QMessageBox.information(self, "Éxito", "Instituciones actualizadas.")
285            except Exception as e:
286                QMessageBox.critical(self, "Error", f"No se pudo actualizar: {e}")

121        """Realiza una prueba de conexión con los parámetros ingresados.

136        password = self.lineEdit_Password.text()
137

139            QMessageBox.warning(self, "Campos Vacíos", "Llena todos los campos.")
140            return
141
142        url_prueba = f"postgresql://{user}:{password}@{host}:{port}/{db_name}"
143        QApplication.setOverrideCursor(Qt.CursorShape.WaitCursor)
144
145        try:
146            motor_prueba = create_engine(url_prueba, connect_args={'connect_timeout': 5})
147            with motor_prueba.connect() as conexion:
148                QApplication.restoreOverrideCursor()
149                QMessageBox.information(self, "¡Éxito!", "Conexión exitosa a la base de datos PostgreSQL.")
150        except Exception as e:
151            QApplication.restoreOverrideCursor()
152            QMessageBox.critical(self, "Error", f"Fallo en conexión:\n{str(e)[:200]}...")
