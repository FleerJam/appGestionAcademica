 52    def __init__(self, parent=None, titulo="Seleccionar", items=None):

 56            parent (QWidget, optional): Widget padre. Defaults to None.

 93    def _cargar_items(self, lista_tuples):
 94        """Carga los items en el widget de lista visual.
 95
 96        Args:
 97            lista_tuples (list): Lista de tuplas (texto_mostrar, data_id).

 99        self.lista_widget.clear()
100        for texto, data_id in lista_tuples:
101            item = QListWidgetItem(texto)
102            item.setData(Qt.ItemDataRole.UserRole, data_id)
103            self.lista_widget.addItem(item)

105    def filtrar_lista(self, texto):
106        """Filtra los elementos visibles de la lista según el texto ingresado.
107
108        Args:
109            texto (str): Texto a buscar (case-insensitive).
110        """
111        texto = texto.lower().strip()
112        for i in range(self.lista_widget.count()):
113            item = self.lista_widget.item(i)
114            item.setHidden(texto not in item.text().lower())

116    def validar_seleccion(self):
117        """Valida y guarda la selección actual antes de cerrar el diálogo.
118
119        Si hay un item seleccionado, guarda su ID y texto en los atributos de la clase.
120        Si solo queda un item visible tras el filtro, lo selecciona automáticamente.
121        Muestra una advertencia si no se selecciona nada.
122        """
123        item = self.lista_widget.currentItem()
124        if not item:
125            visibles = [self.lista_widget.item(i) for i in range(self.lista_widget.count()) if
126                        not self.lista_widget.item(i).isHidden()]
127            if len(visibles) == 1: item = visibles[0]
128
129        if item:
130            self.seleccionado_id = item.data(Qt.ItemDataRole.UserRole)
131            self.seleccionado_texto = item.text()
132            self.accept()
133        else:
134            QMessageBox.warning(self, "Selección requerida", "Por favor seleccione un curso de la lista.")

269    def cargar_datos_tabla(self):
270        """
271        Carga y procesa los datos de matrículas y certificados para mostrar en la tabla.
272
273        Realiza las siguientes acciones:
274        1. Obtiene filtros desde la UI (texto y curso seleccionado).
275        2. Consulta matrículas y certificados en la base de datos.
276        3. Cruza la información para determinar el estado (FIRMADO/GENERADO).
277        4. Calcula contadores de pendientes de firma.
278        5. Aplica paginación y renderiza las filas en el QTableWidget.
279        """
280        if not self.tabla: return
281        self.tabla.setUpdatesEnabled(False)
282        self.tabla.setRowCount(0)
283
284        try:
285            texto = self.input_buscar.text().strip().lower() if self.input_buscar else ""
286            filtros_bd = {"estado": "APROBADO"}
287
288            if self.combo_filtro and self.combo_filtro.currentData():
289                filtros_bd["curso_id"] = self.combo_filtro.currentData()
290
291            matriculas = self.model_matricula.search(filters=filtros_bd) or []
292
293            cert_filtros = {"curso_id": filtros_bd["curso_id"]} if "curso_id" in filtros_bd else None
294            todos_certificados = self.model_certificado.search(filters=cert_filtros) or []
295            mapa_certificados = {(c.persona_id, c.curso_id): c for c in todos_certificados}
296
297            datos_procesados = []
298            cont_faltan_firmar = 0
299
300            for m in matriculas:
301                cert_existente = mapa_certificados.get((m.persona_id, m.curso_id))
302                if not cert_existente: continue
303
304                tiene_firma = bool(m.ruta_pdf_firmado) or bool(cert_existente.ruta_pdf)
305                if not tiene_firma: cont_faltan_firmar += 1
306
307                nombre = m.persona.nombre.lower() if m.persona else ""
308                cedula = m.persona.cedula if m.persona else ""
309                cod = str(cert_existente.codigo_validacion).lower() if cert_existente.codigo_validacion else ""
310
311                if texto and not any(texto in val for val in [nombre, cedula, cod]):
312                    continue
313
314                estado_str = "FIRMADO" if tiene_firma else "GENERADO"
315                datos_procesados.append({
316                    "id_referencia": m.id,
317                    "codigo": cert_existente.codigo_validacion or "---",
318                    "estudiante": m.persona.nombre if m.persona else "---",
319                    "curso": m.curso.nombre if m.curso else "---",
320                    "fecha": str(m.curso.fecha_final) if (m.curso and m.curso.fecha_final) else "-",
321                    "estado": estado_str
322                })
323
324            if self.lbl_num_firmar:
325                self.lbl_num_firmar.setText(f"{cont_faltan_firmar} certificados")
326                style = "color: #e67e22; font-weight: bold;" if cont_faltan_firmar > 0 else "color: #2ecc71;"
327                self.lbl_num_firmar.setStyleSheet(style)
328
329            self.total_registros = len(datos_procesados)
330            self.total_paginas = max(1, (
331                        self.total_registros + self.registros_por_pagina - 1) // self.registros_por_pagina)
332            if self.pagina_actual > self.total_paginas: self.pagina_actual = 1
333
334            inicio = (self.pagina_actual - 1) * self.registros_por_pagina
335            fin = inicio + self.registros_por_pagina
336            lote = datos_procesados[inicio:fin]
337
338            filas_ui = []
339            ids_para_filas = []
340
341            for item in lote:
342                ids_para_filas.append(item["id_referencia"])
343                filas_ui.append(
344                    ["", "", "", "", item["codigo"], item["estudiante"], item["curso"], item["fecha"], item["estado"]])
345
346            PyQtHelper.cargar_datos_en_tabla(self.tabla, filas_ui, alineacion=Qt.AlignmentFlag.AlignCenter)
347
348            limit = min(self.tabla.rowCount(), len(ids_para_filas))
349            for i in range(limit):
350                id_real = ids_para_filas[i]
351                for col in range(4):
352                    item = self.tabla.item(i, col)
353                    if not item:
354                        item = QTableWidgetItem("")
355                        item.setTextAlignment(Qt.AlignmentFlag.AlignCenter)
356                        self.tabla.setItem(i, col, item)
357                    item.setData(Qt.ItemDataRole.UserRole, id_real)
358
359            if self.lbl_contador_sub:
360                self.lbl_contador_sub.setText(f"Mostrando: {len(lote)} de {self.total_registros}")
361            self.actualizar_interfaz_paginacion()
362
363        except Exception as e:
364            traceback.print_exc()
365            QMessageBox.critical(self, "Error Carga", f"Error cargando tabla:\n{str(e)}")
366        finally:
367            self.tabla.setUpdatesEnabled(True)

369    def _get_certificado_by_matricula(self, matricula_id):
370        """Busca la matrícula y el certificado asociado por ID de matrícula.
371
372        Args:
373            matricula_id (int): ID de la matrícula.
374
375        Returns:
376            tuple: (objeto_matricula, objeto_certificado) o (None, None) si falla.
377        """
378        try:
379            if not matricula_id: return None, None
380            mats = self.model_matricula.search(filters={'id': matricula_id})
381            mat = mats[0] if mats else None
382            if not mat: return None, None
383
384            filtros = {"persona_id": mat.persona_id, "curso_id": mat.curso_id}
385            cert = self.model_certificado.search(filters=filtros, first=True)
386            return mat, cert
387        except Exception:
388            return None, None

836                 "NOVIEMBRE", "DICIEMBRE"]

828        """Convierte una fecha a string en formato 'MES DE AÑO'.
