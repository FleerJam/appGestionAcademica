200    def cargar_datos_tabla(self):
201        """Recupera las matrículas de la base de datos y las muestra en la tabla.
202
203        Aplica filtros de búsqueda por cédula, nombre o institución si se ha ingresado
204        texto en el campo de búsqueda. Maneja el estado de "Sin Datos" si no hay registros.
205        """
206        if not self.curso or not self.tabla:
207            return
208
209        # Obtener texto de búsqueda de forma segura
210        texto_busqueda = ""
211        if hasattr(self, 'input_buscar') and self.input_buscar:
212            texto_busqueda = self.input_buscar.text().strip()
213
214        filtros = {'curso_id': self.curso.id}
215        or_fields = []
216
217        if texto_busqueda:
218            or_fields = [
219                ('persona_cedula', texto_busqueda),
220                ('persona_nombre', texto_busqueda),
221                ('persona_institucion_articulada', texto_busqueda)
222            ]
223
224        try:
225            self.total_registros = self.matricula_model.count(
226                filters=filtros,
227                or_fields=or_fields if or_fields else None
228            )
229
230            if hasattr(self, 'lbl_contador'):
231                self.lbl_contador.setText(f"Total matrículas: {self.total_registros}")
232
233            # --- CASO SIN DATOS ---
234            if self.total_registros == 0:
235                self.tabla.clearContents()
236                self.tabla.setRowCount(0)
237
238                # Ocultar headers si no hay datos
239                self.tabla.horizontalHeader().setVisible(False)
240
241                if hasattr(self, 'lbl_sin_datos'):
242                    self.lbl_sin_datos.setVisible(True)
243                return
244
245            # --- CASO CON DATOS ---
246            if hasattr(self, 'lbl_sin_datos'):
247                self.lbl_sin_datos.setVisible(False)
248
249            # Restaurar visibilidad de headers
250            self.tabla.horizontalHeader().setVisible(True)
251
252            matriculas = self.matricula_model.search(
253                filters=filtros,
254                order_by="persona_nombre",
255                or_fields=or_fields if or_fields else None
256            )
257
258        except Exception as e:
259            print(f"Error cargando matrículas: {e}")
260            return
261
262        filas = []
263        for matricula in matriculas:
264            persona = matricula.persona
265
266            txt_centro = "Sin Asignar"
267            if matricula.centro_id:
268                try:
269                    txt_centro = self.model_centro.texto_desde_id(matricula.centro_id)
270                except Exception:
271                    pass
272
273            filas.append([
274                matricula.id,
275                persona.cedula if persona else "-",
276                persona.nombre if persona else "-",
277                persona.correo if persona else "-",
278                txt_centro,
279                persona.institucion_articulada if persona else "-"
280            ])
281
282        # Recargamos configuración (incluye textos de headers)
283        self.cargar_formato_tabla()
284
285        PyQtHelper.cargar_datos_en_tabla(
286            self.tabla,
287            filas,
288            id_column=0,
289            alineacion=Qt.AlignmentFlag.AlignCenter
290        )

292    def _cargar_datos_curso(self):
293        """Rellena los campos del formulario con la información actual del curso."""
294        if not self.curso: return
295        try:
296            self.input_nombre.setText(self.curso.nombre.lower().title())
297            self.input_responsable.setText(self.curso.responsable.lower().title())
298            self.input_duracion.setValue(self.curso.duracion_horas)
299            self.input_nota.setValue(float(self.curso.nota_aprobacion))
300            self.input_tipo.setCurrentText(self.curso.tipo_curso.lower().title())
301            self.input_modalidad.setCurrentText(self.curso.modalidad.lower().title())
302
303            fi = self.curso.fecha_inicio
304            ff = self.curso.fecha_final
305            self.date_inicio.setDate(QDate(fi.year, fi.month, fi.day))
306            self.date_fin.setDate(QDate(ff.year, ff.month, ff.day))
307
308            participantes = self.curso.participantes_objetivo or []
309            if isinstance(participantes, str):
310                limpio = participantes.replace('[', '').replace(']', '').replace('"', '').replace("'", "")
311                participantes = [p.strip() for p in limpio.split(',') if p.strip()]
312
313            participantes_norm = {str(p).strip().upper() for p in participantes}
314
315            self.checkBox_sis.setChecked("PERSONAL DEL SIS ECU 911" in participantes_norm)
316            self.checkBox_instituciones.setChecked("INSTITUCIONES ARTICULADAS Y/O VINCULADAS" in participantes_norm)
317            self.checkBox_ciudadania.setChecked("CIUDADANIA EN GENERAL" in participantes_norm)
318        except Exception as e:
319            print(f"Error cargando datos de BD al formulario: {e}")

351    def guardar_cambios(self):
352        """Valida los datos ingresados y actualiza el curso en la base de datos.
353
354        Realiza validaciones de fechas, tipos de datos y selección de participantes.
355        Si la actualización es exitosa, recalcula estados de matrículas y refresca la UI.
356        """
357        try:
358            duracion = self.input_duracion.value()
359            nota_aprobacion = self.input_nota.value()
360        except (TypeError, ValueError):
361            self.mostrar_mensaje("error", "Error", "Valores numéricos inválidos.")
362            return
363
364        fecha_inicio_py = self.date_inicio.date().toPyDate()
365        fecha_final_py = self.date_fin.date().toPyDate()
366
367        if fecha_final_py < fecha_inicio_py:
368            self.mostrar_mensaje("error", "Error", "La fecha final no puede ser anterior a la inicial.")
369            return
370
371        participantes_seleccionados = []
372        if self.checkBox_sis.isChecked():
373            participantes_seleccionados.append("PERSONAL DEL SIS ECU 911")
374        if self.checkBox_instituciones.isChecked():
375            participantes_seleccionados.append("INSTITUCIONES ARTICULADAS Y/O VINCULADAS")
376        if self.checkBox_ciudadania.isChecked():
377            participantes_seleccionados.append("CIUDADANIA EN GENERAL")
378
379        if not participantes_seleccionados:
380            self.mostrar_mensaje("error", "Error", "Debe seleccionar al menos un tipo de participante.")
381            return
382
383        datos_actualizados = {
384            "nombre": self.input_nombre.text().strip().upper(),
385            "responsable": self.input_responsable.text().strip().upper(),
386            "duracion_horas": duracion,
387            "tipo_curso": self.input_tipo.currentText().strip().upper(),
388            "modalidad": self.input_modalidad.currentText().strip().upper(),
389            "fecha_inicio": fecha_inicio_py,
390            "fecha_final": fecha_final_py,
391            "nota_aprobacion": nota_aprobacion,
392            "participantes_objetivo": participantes_seleccionados
393        }
394
395        try:
396            exito = self.curso_model.update(self.curso.id, datos_actualizados)
397
398            if exito:
399                self.curso = self.curso_model.get_by_id(self.curso.id)
400
401                # --- NUEVA LÓGICA DE ACTUALIZACIÓN DE ESTADOS ---
402                # Si cambian las reglas (nota mínima o fechas), recalcular estados.
403                self.matricula_model.actualizar_estados_por_curso(curso=self.curso)
404
405                # --- ACTUALIZAR MASTER (DASHBOARD) ---
406                self._actualizar_master()
407
408                self.mostrar_mensaje("info", "Éxito", "El curso ha sido actualizado correctamente.")
409                self.signal_actualizado.emit()
410
411                self._bloquear_campos()
412                self.btn_editar.setText("Editar")
413                self.btn_editar.clicked.disconnect()
414                self.btn_editar.clicked.connect(self.activar_edicion)
415            else:
416                self.mostrar_mensaje("error", "Error", "No se encontró el registro para actualizar.")
417
418        except IntegrityError:
419            self.mostrar_mensaje("error", "Error", "Ya existe un curso con datos duplicados.")
420        except Exception as e:
421            self.mostrar_mensaje("error", "Error inesperado", str(e))

280                persona.institucion_articulada if persona else "-"

353        """Valida los datos ingresados y actualiza el curso en la base de datos.
354
355        Realiza validaciones de fechas, tipos de datos y selección de participantes.
356        Si la actualización es exitosa, recalcula estados de matrículas y refresca la UI.
357        """
358        try:
359            duracion = self.input_duracion.value()
360            nota_aprobacion = self.input_nota.value()
361        except (TypeError, ValueError):
362            self.mostrar_mensaje("error", "Error", "Valores numéricos inválidos.")
363            return
364
365        fecha_inicio_py = self.date_inicio.date().toPyDate()
366        fecha_final_py = self.date_fin.date().toPyDate()
367
368        if fecha_final_py < fecha_inicio_py:
369            self.mostrar_mensaje("error", "Error", "La fecha final no puede ser anterior a la inicial.")
370            return
371
372        participantes_seleccionados = []
373        if self.checkBox_sis.isChecked():
374            participantes_seleccionados.append("PERSONAL DEL SIS ECU 911")
375        if self.checkBox_instituciones.isChecked():
376            participantes_seleccionados.append("INSTITUCIONES ARTICULADAS Y/O VINCULADAS")
377        if self.checkBox_ciudadania.isChecked():
378            participantes_seleccionados.append("CIUDADANIA EN GENERAL")
379
380        if not participantes_seleccionados:
381            self.mostrar_mensaje("error", "Error", "Debe seleccionar al menos un tipo de participante.")
382            return
383
384        datos_actualizados = {
385            "nombre": self.input_nombre.text().strip().upper(),
386            "responsable": self.input_responsable.text().strip().upper(),
387            "duracion_horas": duracion,
388            "tipo_curso": self.input_tipo.currentText().strip().upper(),
389            "modalidad": self.input_modalidad.currentText().strip().upper(),
390            "fecha_inicio": fecha_inicio_py,
391            "fecha_final": fecha_final_py,
392            "nota_aprobacion": nota_aprobacion,
393            "participantes_objetivo": participantes_seleccionados
394        }
395
396        try:
397            exito = self.curso_model.update(self.curso.id, datos_actualizados)
398
399            if exito:
400                self.curso = self.curso_model.get_by_id(self.curso.id)
401
402                # --- NUEVA LÓGICA DE ACTUALIZACIÓN DE ESTADOS ---
403                # Si cambian las reglas (nota mínima o fechas), recalcular estados.
404                self.matricula_model.actualizar_estados_por_curso(curso=self.curso)
405
406                # --- ACTUALIZAR MASTER (DASHBOARD) ---
407                self._actualizar_master()
408
409                self.mostrar_mensaje("info", "Éxito", "El curso ha sido actualizado correctamente.")
410                self.signal_actualizado.emit()
411
412                self._bloquear_campos()
413                self.btn_editar.setText("Editar")
414                self.btn_editar.clicked.disconnect()
415                self.btn_editar.clicked.connect(self.activar_edicion)
416            else:
417                self.mostrar_mensaje("error", "Error", "No se encontró el registro para actualizar.")
418
419        except IntegrityError:
420            self.mostrar_mensaje("error", "Error", "Ya existe un curso con datos duplicados.")
421        except Exception as e:
422            self.mostrar_mensaje("error", "Error inesperado", str(e))

352    def guardar_cambios(self):
353        """Valida los datos ingresados y actualiza el curso en la base de datos.
354
355        Realiza validaciones de fechas, tipos de datos y selección de participantes.
356        Si la actualización es exitosa, recalcula estados de matrículas y refresca la UI.

358        try:
359            duracion = self.input_duracion.value()
360            nota_aprobacion = self.input_nota.value()

365        fecha_inicio_py = self.date_inicio.date().toPyDate()
366        fecha_final_py = self.date_fin.date().toPyDate()
367
368        if fecha_final_py < fecha_inicio_py:
369            self.mostrar_mensaje("error", "Error", "La fecha final no puede ser anterior a la inicial.")

378            participantes_seleccionados.append("CIUDADANIA EN GENERAL")
379
380        if not participantes_seleccionados:
381            self.mostrar_mensaje("error", "Error", "Debe seleccionar al menos un tipo de participante.")
382            return
383
384        datos_actualizados = {
385            "nombre": self.input_nombre.text().strip().upper(),
386            "responsable": self.input_responsable.text().strip().upper(),
387            "duracion_horas": duracion,
388            "tipo_curso": self.input_tipo.currentText().strip().upper(),

393            "participantes_objetivo": participantes_seleccionados

421        except Exception as e:
