264    def _cargar_datos_estudiante(self):
265        """Rellena los campos del formulario con los datos del objeto estudiante."""
266        if not self.estudiante:
267            return
268
269        # Cargar textos básicos
270        if self.input_nombre:
271            self.input_nombre.setText(getattr(self.estudiante, 'nombre', '') or '')
272        if self.input_cedula:
273            self.input_cedula.setText(getattr(self.estudiante, 'cedula', '') or '')
274        if self.input_correo:
275            self.input_correo.setText(getattr(self.estudiante, 'correo', '') or '')
276
277        # Helper interno para seleccionar texto en combo insensible a mayúsculas
278        # Esto es vital porque en DB se guardan UPPER (ej. "ESTUDIANTE")
279        # pero el combo tiene Title Case (ej. "Estudiante")
280        def set_combo_text_insensitive(combo, text_value):
281            if not combo or not text_value: return
282            text_value = str(text_value).upper()
283            for i in range(combo.count()):
284                if combo.itemText(i).upper() == text_value:
285                    combo.setCurrentIndex(i)
286                    return
287
288        # Seleccionar ROL
289        rol_actual = getattr(self.estudiante, 'rol', 'Estudiante')
290        set_combo_text_insensitive(self.input_rol, rol_actual)
291
292        # Seleccionar INSTITUCION
293        inst_actual = getattr(self.estudiante, 'institucion_articulada', '')
294        set_combo_text_insensitive(self.input_institucion, inst_actual)
295
296
297        # Seleccionar CENTRO (por ID)
298        if self.input_centro:
299            centro_id_actual = getattr(self.estudiante, 'centro_id', None)
300            if centro_id_actual:
301                index = self.input_centro.findData(centro_id_actual)
302                if index >= 0:
303                    self.input_centro.setCurrentIndex(index)
304            else:
305                self.input_centro.setCurrentIndex(0)

317    def guardar_cambios(self):
318        """Valida los datos y actualiza el registro del estudiante.
319
320        Realiza validaciones de campos obligatorios, formato de cédula y correo.
321        Maneja errores de integridad (duplicados).
322        """
323        # Recolección de datos
324        nombre = self.input_nombre.text().strip().upper() if self.input_nombre else ""
325        cedula = self.input_cedula.text().strip() if self.input_cedula else ""
326        correo = self.input_correo.text().strip().lower() if self.input_correo else ""
327
328        # OBTENCIÓN DE DATOS DE COMBOS (Con .upper() para consistencia en BD)
329        rol = self.input_rol.currentText().strip().upper() if self.input_rol else "ESTUDIANTE"
330        institucion = self.input_institucion.currentText().strip().upper() if self.input_institucion else None
331
332        centro_id = self.input_centro.currentData() if self.input_centro else None
333
334        # --- VALIDACIONES (Igual que en DialogoPersona) ---
335
336        # 1. Campos obligatorios
337        if not cedula or not nombre or not correo or not rol or not centro_id or not institucion:
338            QMessageBox.warning(self, "Error", "Todos los campos son obligatorios.")
339            return
340
341        # 2. Validación de Cédula Ecuatoriana
342        if not self.sanitizier.validar_cedula_ecuador(cedula):
343            QMessageBox.warning(self, "Error", "Cédula inválida.")
344            return
345
346        # 3. Validación de formato de correo
347        patron_correo = r'^[\w\.-]+@[\w\.-]+\.\w+$'
348        if not re.match(patron_correo, correo):
349            QMessageBox.warning(self, "Error", "El formato del correo electrónico no es válido.")
350            return
351
352        datos_actualizados = {
353            "nombre": nombre,
354            "cedula": cedula,
355            "correo": correo,
356            "rol": rol,
357            "centro_id": centro_id,
358            "institucion_articulada": institucion
359        }
360
361        try:
362            exito = self.persona_model.update(self.estudiante.id, datos_actualizados)
363
364            if exito:
365                self.estudiante = self.persona_model.get_by_id(self.estudiante.id)
366                QMessageBox.information(self, "Éxito", "Datos actualizados correctamente.")
367                self.signal_actualizado.emit()
368
369                self._bloquear_campos()
370                self.btn_editar.setText("Editar")
371                self.btn_editar.clicked.disconnect()
372                self.btn_editar.clicked.connect(self.activar_edicion)
373            else:
374                QMessageBox.warning(self, "Error", "No se encontró el registro para actualizar.")
375
376        except IntegrityError:
377            QMessageBox.critical(self, "Error", "La cédula o el correo ya existen en otro registro.")
378        except Exception as e:
379            QMessageBox.critical(self, "Error", str(e))

400    def cargar_datos_tabla(self):
401        """Obtiene y visualiza el historial de matrículas del estudiante.
402
403        Muestra los cursos en los que está inscrito, aplicando filtros si existen.
404        Maneja la carga diferida de relaciones (curso, centro) para evitar errores de sesión.
405        """
406        if not self.tabla or not self.estudiante: return
407        try:
408            texto_busqueda = self.input_buscar.text().strip() if self.input_buscar else ""
409
410            filtros = {'persona_id': self.estudiante.id}
411            or_fields = []
412
413            if texto_busqueda:
414                or_fields = [
415                    ('curso_nombre', texto_busqueda),
416                    ('curso_modalidad', texto_busqueda)
417                ]
418
419            try:
420                self.total_registros = self.matricula_model.count(
421                    filters=filtros,
422                    or_fields=or_fields if or_fields else None
423                )
424
425                if hasattr(self, 'lbl_contador'):
426                    self.lbl_contador.setText(f"Cursos inscritos: {self.total_registros}")
427
428                if self.total_registros == 0:
429                    self.tabla.clearContents()
430                    self.tabla.setRowCount(0)
431                    self.tabla.horizontalHeader().setVisible(False)
432                    if hasattr(self, 'lbl_sin_datos'):
433                        self.lbl_sin_datos.setVisible(True)
434                    return
435
436                if hasattr(self, 'lbl_sin_datos'):
437                    self.lbl_sin_datos.setVisible(False)
438
439                self.tabla.horizontalHeader().setVisible(True)
440
441                matriculas = self.matricula_model.search(
442                    filters=filtros,
443                    order_by="curso_nombre",
444                    or_fields=or_fields if or_fields else None
445                )
446
447            except Exception as e:
448                print(f"Error cargando tabla matrículas: {e}")
449                return
450
451            self.cargar_formato_tabla()
452
453            filas = []
454            for m in matriculas:
455                # FIX: Detached Instance Error en 'm.curso'
456                # La sesión se cerró al terminar .search(), por lo que la relación m.curso
457                # no puede cargarse (Lazy Load). Usamos el curso_id para obtenerlo explícitamente.
458                curso = None
459                curso_id = getattr(m, 'curso_id', None)
460                if curso_id:
461                    try:
462                        curso = self.curso_model.get_by_id(curso_id)
463                    except Exception:
464                        curso = None
465
466                centro = "Sin Asignar"
467
468                if m.centro_id:
469                    try:
470                        centro = self.centro_model.texto_desde_id(m.centro_id)
471                    except:
472                        pass
473
474                filas.append([
475                    m.id,
476                    curso.nombre if curso else "-",
477                    curso.tipo_curso if curso else "-",
478                    curso.modalidad if curso else "-",
479                    f"{curso.duracion_horas} hrs" if curso else "-",
480                    centro
481                ])
482
483            PyQtHelper.cargar_datos_en_tabla(
484                self.tabla,
485                filas,
486                id_column=0,
487                alineacion=Qt.AlignmentFlag.AlignCenter
488            )
489        except Exception as e:
490            print(e)

265    def _cargar_datos_estudiante(self):
266        """Rellena los campos del formulario con los datos del objeto estudiante."""
267        if not self.estudiante:
268            return
269
270        # Cargar textos básicos
271        if self.input_nombre:
272            self.input_nombre.setText(getattr(self.estudiante, 'nombre', '') or '')
273        if self.input_cedula:
274            self.input_cedula.setText(getattr(self.estudiante, 'cedula', '') or '')
275        if self.input_correo:
276            self.input_correo.setText(getattr(self.estudiante, 'correo', '') or '')
277
278        # Helper interno para seleccionar texto en combo insensible a mayúsculas
279        # Esto es vital porque en DB se guardan UPPER (ej. "ESTUDIANTE")
280        # pero el combo tiene Title Case (ej. "Estudiante")
281        def set_combo_text_insensitive(combo, text_value):
282            if not combo or not text_value: return
283            text_value = str(text_value).upper()
284            for i in range(combo.count()):
285                if combo.itemText(i).upper() == text_value:
286                    combo.setCurrentIndex(i)
287                    return
288
289        # Seleccionar ROL
290        rol_actual = getattr(self.estudiante, 'rol', 'Estudiante')
291        set_combo_text_insensitive(self.input_rol, rol_actual)
292
293        # Seleccionar INSTITUCION
294        inst_actual = getattr(self.estudiante, 'institucion_articulada', '')
295        set_combo_text_insensitive(self.input_institucion, inst_actual)
296
297
298        # Seleccionar CENTRO (por ID)
299        if self.input_centro:
300            centro_id_actual = getattr(self.estudiante, 'centro_id', None)
301            if centro_id_actual:
302                index = self.input_centro.findData(centro_id_actual)
303                if index >= 0:
304                    self.input_centro.setCurrentIndex(index)
305            else:
306                self.input_centro.setCurrentIndex(0)

318    def guardar_cambios(self):
319        """Valida los datos y actualiza el registro del estudiante.
320
321        Realiza validaciones de campos obligatorios, formato de cédula y correo.
322        Maneja errores de integridad (duplicados).
323        """
324        # Recolección de datos
325        nombre = self.input_nombre.text().strip().upper() if self.input_nombre else ""
326        cedula = self.input_cedula.text().strip() if self.input_cedula else ""
327        correo = self.input_correo.text().strip().lower() if self.input_correo else ""
328
329        # OBTENCIÓN DE DATOS DE COMBOS (Con .upper() para consistencia en BD)
330        rol = self.input_rol.currentText().strip().upper() if self.input_rol else "ESTUDIANTE"
331        institucion = self.input_institucion.currentText().strip().upper() if self.input_institucion else None
332
333        centro_id = self.input_centro.currentData() if self.input_centro else None
334
335        # --- VALIDACIONES (Igual que en DialogoPersona) ---
336
337        # 1. Campos obligatorios
338        if not cedula or not nombre or not correo or not rol or not centro_id or not institucion:
339            QMessageBox.warning(self, "Error", "Todos los campos son obligatorios.")
340            return
341
342        # 2. Validación de Cédula Ecuatoriana
343        if not self.sanitizier.validar_cedula_ecuador(cedula):
344            QMessageBox.warning(self, "Error", "Cédula inválida.")
345            return
346
347        # 3. Validación de formato de correo
348        patron_correo = r'^[\w\.-]+@[\w\.-]+\.\w+$'
349        if not re.match(patron_correo, correo):
350            QMessageBox.warning(self, "Error", "El formato del correo electrónico no es válido.")
351            return
352
353        datos_actualizados = {
354            "nombre": nombre,
355            "cedula": cedula,
356            "correo": correo,
357            "rol": rol,
358            "centro_id": centro_id,
359            "institucion_articulada": institucion
360        }
361
362        try:
363            exito = self.persona_model.update(self.estudiante.id, datos_actualizados)
364
365            if exito:
366                self.estudiante = self.persona_model.get_by_id(self.estudiante.id)
367                QMessageBox.information(self, "Éxito", "Datos actualizados correctamente.")
368                self.signal_actualizado.emit()
369
370                self._bloquear_campos()
371                self.btn_editar.setText("Editar")
372                self.btn_editar.clicked.disconnect()
373                self.btn_editar.clicked.connect(self.activar_edicion)
374            else:
375                QMessageBox.warning(self, "Error", "No se encontró el registro para actualizar.")
376
377        except IntegrityError:
378            QMessageBox.critical(self, "Error", "La cédula o el correo ya existen en otro registro.")
379        except Exception as e:
380            QMessageBox.critical(self, "Error", str(e))

401    def cargar_datos_tabla(self):
402        """Obtiene y visualiza el historial de matrículas del estudiante.
403
404        Muestra los cursos en los que está inscrito, aplicando filtros si existen.
405        Maneja la carga diferida de relaciones (curso, centro) para evitar errores de sesión.
406        """
407        if not self.tabla or not self.estudiante: return
408        try:
409            texto_busqueda = self.input_buscar.text().strip() if self.input_buscar else ""
410
411            filtros = {'persona_id': self.estudiante.id}
412            or_fields = []
413
414            if texto_busqueda:
415                or_fields = [
416                    ('curso_nombre', texto_busqueda),
417                    ('curso_modalidad', texto_busqueda)
418                ]
419
420            try:
421                self.total_registros = self.matricula_model.count(
422                    filters=filtros,
423                    or_fields=or_fields if or_fields else None
424                )
425
426                if hasattr(self, 'lbl_contador'):
427                    self.lbl_contador.setText(f"Cursos inscritos: {self.total_registros}")
428
429                if self.total_registros == 0:
430                    self.tabla.clearContents()
431                    self.tabla.setRowCount(0)
432                    self.tabla.horizontalHeader().setVisible(False)
433                    if hasattr(self, 'lbl_sin_datos'):
434                        self.lbl_sin_datos.setVisible(True)
435                    return
436
437                if hasattr(self, 'lbl_sin_datos'):
438                    self.lbl_sin_datos.setVisible(False)
439
440                self.tabla.horizontalHeader().setVisible(True)
441
442                matriculas = self.matricula_model.search(
443                    filters=filtros,
444                    order_by="curso_nombre",
445                    or_fields=or_fields if or_fields else None
446                )
447
448            except Exception as e:
449                print(f"Error cargando tabla matrículas: {e}")
450                return
451
452            self.cargar_formato_tabla()
453
454            filas = []
455            for m in matriculas:
456                # FIX: Detached Instance Error en 'm.curso'
457                # La sesión se cerró al terminar .search(), por lo que la relación m.curso
458                # no puede cargarse (Lazy Load). Usamos el curso_id para obtenerlo explícitamente.
459                curso = None
460                curso_id = getattr(m, 'curso_id', None)
461                if curso_id:
462                    try:
463                        curso = self.curso_model.get_by_id(curso_id)
464                    except Exception:
465                        curso = None
466
467                centro = "Sin Asignar"
468
469                if m.centro_id:
470                    try:
471                        centro = self.centro_model.texto_desde_id(m.centro_id)
472                    except:
473                        pass
474
475                filas.append([
476                    m.id,
477                    curso.nombre if curso else "-",
478                    curso.tipo_curso if curso else "-",
479                    curso.modalidad if curso else "-",
480                    f"{curso.duracion_horas} hrs" if curso else "-",
481                    centro
482                ])
483
484            PyQtHelper.cargar_datos_en_tabla(
485                self.tabla,
486                filas,
487                id_column=0,
488                alineacion=Qt.AlignmentFlag.AlignCenter
489            )
490        except Exception as e:
491            print(e)

def guardar_cambios(self): 
318    def guardar_cambios(self):
319        """Valida los datos y actualiza el registro del estudiante.
320
321        Realiza validaciones de campos obligatorios, formato de cédula y correo.
322        Maneja errores de integridad (duplicados).
323        """
324        # Recolección de datos
325        nombre = self.input_nombre.text().strip().upper() if self.input_nombre else ""
326        cedula = self.input_cedula.text().strip() if self.input_cedula else ""
327        correo = self.input_correo.text().strip().lower() if self.input_correo else ""
328
329        # OBTENCIÓN DE DATOS DE COMBOS (Con .upper() para consistencia en BD)
330        rol = self.input_rol.currentText().strip().upper() if self.input_rol else "ESTUDIANTE"
331        institucion = self.input_institucion.currentText().strip().upper() if self.input_institucion else None
332
333        centro_id = self.input_centro.currentData() if self.input_centro else None
334
335        # --- VALIDACIONES (Igual que en DialogoPersona) ---
336
337        # 1. Campos obligatorios
338        if not cedula or not nombre or not correo or not rol or not centro_id or not institucion:
339            QMessageBox.warning(self, "Error", "Todos los campos son obligatorios.")
340            return
341
342        # 2. Validación de Cédula Ecuatoriana
343        if not self.sanitizier.validar_cedula_ecuador(cedula):
344            QMessageBox.warning(self, "Error", "Cédula inválida.")
345            return
346
347        # 3. Validación de formato de correo
348        patron_correo = r'^[\w\.-]+@[\w\.-]+\.\w+$'
349        if not re.match(patron_correo, correo):
350            QMessageBox.warning(self, "Error", "El formato del correo electrónico no es válido.")
351            return
352
353        datos_actualizados = {
354            "nombre": nombre,
355            "cedula": cedula,
356            "correo": correo,
357            "rol": rol,
358            "centro_id": centro_id,
359            "institucion_articulada": institucion
360        }
361
362        try:
363            exito = self.persona_model.update(self.estudiante.id, datos_actualizados)
364
365            if exito:
366                self.estudiante = self.persona_model.get_by_id(self.estudiante.id)
367                QMessageBox.information(self, "Éxito", "Datos actualizados correctamente.")
368                self.signal_actualizado.emit()
369
370                self._bloquear_campos()
371                self.btn_editar.setText("Editar")
372                self.btn_editar.clicked.disconnect()
373                self.btn_editar.clicked.connect(self.activar_edicion)
374            else:
375                QMessageBox.warning(self, "Error", "No se encontró el registro para actualizar.")
376
377        except IntegrityError:
378            QMessageBox.critical(self, "Error", "La cédula o el correo ya existen en otro registro.")
379        except Exception as e:
380            QMessageBox.critical(self, "Error", str(e))

def cargar_datos_tabla(self): 
401    def cargar_datos_tabla(self):
402        """Obtiene y visualiza el historial de matrículas del estudiante.
403
404        Muestra los cursos en los que está inscrito, aplicando filtros si existen.
405        Maneja la carga diferida de relaciones (curso, centro) para evitar errores de sesión.
406        """
407        if not self.tabla or not self.estudiante: return
408        try:
409            texto_busqueda = self.input_buscar.text().strip() if self.input_buscar else ""
410
411            filtros = {'persona_id': self.estudiante.id}
412            or_fields = []
413
414            if texto_busqueda:
415                or_fields = [
416                    ('curso_nombre', texto_busqueda),
417                    ('curso_modalidad', texto_busqueda)
418                ]
419
420            try:
421                self.total_registros = self.matricula_model.count(
422                    filters=filtros,
423                    or_fields=or_fields if or_fields else None
424                )
425
426                if hasattr(self, 'lbl_contador'):
427                    self.lbl_contador.setText(f"Cursos inscritos: {self.total_registros}")
428
429                if self.total_registros == 0:
430                    self.tabla.clearContents()
431                    self.tabla.setRowCount(0)
432                    self.tabla.horizontalHeader().setVisible(False)
433                    if hasattr(self, 'lbl_sin_datos'):
434                        self.lbl_sin_datos.setVisible(True)
435                    return
436
437                if hasattr(self, 'lbl_sin_datos'):
438                    self.lbl_sin_datos.setVisible(False)
439
440                self.tabla.horizontalHeader().setVisible(True)
441
442                matriculas = self.matricula_model.search(
443                    filters=filtros,
444                    order_by="curso_nombre",
445                    or_fields=or_fields if or_fields else None
446                )
447
448            except Exception as e:
449                print(f"Error cargando tabla matrículas: {e}")
450                return
451
452            self.cargar_formato_tabla()
453
454            filas = []
455            for m in matriculas:
456                # FIX: Detached Instance Error en 'm.curso'
457                # La sesión se cerró al terminar .search(), por lo que la relación m.curso
458                # no puede cargarse (Lazy Load). Usamos el curso_id para obtenerlo explícitamente.
459                curso = None
460                curso_id = getattr(m, 'curso_id', None)
461                if curso_id:
462                    try:
463                        curso = self.curso_model.get_by_id(curso_id)
464                    except Exception:
465                        curso = None
466
467                centro = "Sin Asignar"
468
469                if m.centro_id:
470                    try:
471                        centro = self.centro_model.texto_desde_id(m.centro_id)
472                    except:
473                        pass
474
475                filas.append([
476                    m.id,
477                    curso.nombre if curso else "-",
478                    curso.tipo_curso if curso else "-",
479                    curso.modalidad if curso else "-",
480                    f"{curso.duracion_horas} hrs" if curso else "-",
481                    centro
482                ])
483
484            PyQtHelper.cargar_datos_en_tabla(
485                self.tabla,
486                filas,
487                id_column=0,
488                alineacion=Qt.AlignmentFlag.AlignCenter
489            )
490        except Exception as e:
491            print(e)
