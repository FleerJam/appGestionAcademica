 23class Centro(Base):
 24    """Modelo que representa un centro operativo o administrativo (Zonal, Local, Sala)."""
 25    __tablename__ = "centros"
 26
 27    id = Column(String(26), primary_key=True, default=generar_uid, index=True)
 28    nombre = Column(String(150), nullable=False)

 90class PlantillaCertificado(Base):
 91    """
 92    Define el diseo visual de un certificado.
 93    Puede almacenar la configuracin JSON para renderizado canvas o el binario de Word.
 94    """
 95    __tablename__ = "plantillas_certificado"
 96
 97    id = Column(String(26), primary_key=True, default=generar_uid, index=True)
 98    nombre = Column(String(150), nullable=False)  # Ej: "Plantilla Oficial 2026"
 99
100    # Configuracin del Lienzo / Fondo
101    ancho_px = Column(Integer, default=1123)  # A4 Horizontal aprox
102    alto_px = Column(Integer, default=794)
103    numero_paginas = Column(Integer, default=1)
104
105    # El JSON generado por el diseador (coordenadas, fuentes, variables)
106    configuracion_json = Column(JSON, nullable=False)
107    archivo_binario = Column(LargeBinary, nullable=True)
108    # Relacin inversa (opcional, para ver qu cursos usan esta plantilla)
109    cursos = relationship("Curso", back_populates="plantilla")

112class Curso(Base):
113    """Modelo que representa un evento de capacitacin o adiestramiento."""
114    __tablename__ = "cursos"
115
116    id = Column(String(26), primary_key=True, default=generar_uid, index=True)
117    nombre = Column(String(255), nullable=False)

121    responsable = Column(String(255), nullable=False)
122    participantes_objetivo = Column(JSON, nullable=False)
123    duracion_horas = Column(Integer, nullable=False)
124    modalidad = Column(String(100), nullable=False)
125    nota_aprobacion = Column(Float, default=7.0, nullable=False)
126
127    # NUEVO: Vinculacin con la plantilla de certificado
128    plantilla_id = Column(
129        String(26),
130        ForeignKey("plantillas_certificado.id", ondelete="SET NULL"),
131        nullable=True
132    )
133
134    plantilla = relationship("PlantillaCertificado", back_populates="cursos")
135    matriculas = relationship("Matricula", back_populates="curso", cascade="all, delete-orphan")
136    certificados = relationship("Certificado", back_populates="curso", cascade="all, delete-orphan")
137    evaluaciones = relationship("EvaluacionCurso", back_populates="curso", cascade="all, delete-orphan")
