 97    def _forzar_sanitizacion(self, widget: QLineEdit):
 98        """
 99        Método auxiliar para limpiar el texto de un QLineEdit en tiempo real.
100        Evita tildes y fuerza mayúsculas usando Sanitizer.
101        """
102        texto_original = widget.text()
103        if not texto_original:
104            return
105
106        # Usamos el sanitizer provisto
107        texto_limpio = Sanitizer.casi_limpio(texto_original)
108
109        if texto_original != texto_limpio:
110            # Guardamos posición del cursor para no incomodar al usuario
111            pos = widget.cursorPosition()
112            widget.setText(texto_limpio)
113            # Restauramos cursor (ajustando si la longitud cambió, aunque limpiar_texto suele mantenerla)
114            widget.setCursorPosition(pos)

118    def cargar_esquema(self):
119        """
120        Carga las evaluaciones existentes desde la BD y las puebla en la tabla.
121        """
122        self.tabla.setRowCount(0)
123        total_pct = 0.0
124
125        try:
126            self.db.expire_all()
127            evaluaciones = self.db.query(EvaluacionCurso).filter_by(curso_id=self.curso_id).order_by(
128                EvaluacionCurso.orden).all()
129
130            for ev in evaluaciones:
131                row = self.tabla.rowCount()
132                self.tabla.insertRow(row)
133
134                # --- COLUMNA 0: NOMBRE EDITABLE ---
135                # Ahora usamos un QLineEdit en lugar de un QTableWidgetItem estático
136                input_nombre_existente = QLineEdit(ev.nombre)
137                # 1. Sanitizar al escribir (No permite tildes)
138                input_nombre_existente.textChanged.connect(
139                    lambda _, w=input_nombre_existente: self._forzar_sanitizacion(w)
140                )
141                # 2. Guardar en BD al terminar de editar (perder foco o Enter)
142                input_nombre_existente.editingFinished.connect(
143                    lambda eid=ev.id, w=input_nombre_existente: self.actualizar_nombre_bd(eid, w)
144                )
145                self.tabla.setCellWidget(row, 0, input_nombre_existente)
146
147                # --- COLUMNA 1: PORCENTAJE ---
148                spin_pct = QDoubleSpinBox()
149                spin_pct.setRange(0.1, 100.0)
150                spin_pct.setSuffix(" %")
151                spin_pct.setValue(ev.porcentaje)
152                spin_pct.valueChanged.connect(lambda val, eid=ev.id: self.actualizar_porcentaje_bd(eid, val))
153                self.tabla.setCellWidget(row, 1, spin_pct)
154
155                # --- COLUMNA 2: BORRAR ---
156                btn_borrar = QPushButton("Eliminar")
157                btn_borrar.setStyleSheet("color: red;")
158                btn_borrar.clicked.connect(lambda _, eid=ev.id: self.eliminar_evaluacion(eid))
159                self.tabla.setCellWidget(row, 2, btn_borrar)
160
161                total_pct += ev.porcentaje
162
163            self.actualizar_label_total(total_pct)
164
165        except Exception as e:
166            self.mostrar_error(f"Error cargando esquema: {e}")

168    def actualizar_nombre_bd(self, eval_id, widget: QLineEdit):
169        """Actualiza el nombre de la evaluación en la BD cuando se edita en la tabla."""
170        nuevo_nombre = widget.text().strip()
171
172        # Validar que no quede vacío
173        if not nuevo_nombre:
174            self.mostrar_error("El nombre de la actividad no puede estar vacío.")
175            # Restaurar el valor anterior desde la BD
176            try:
177                ev = self.db.query(EvaluacionCurso).get(eval_id)
178                if ev: widget.setText(ev.nombre)
179            except:
180                pass
181            return
182
183        try:
184            ev = self.db.query(EvaluacionCurso).get(eval_id)
185            if ev and ev.nombre != nuevo_nombre:
186                ev.nombre = nuevo_nombre
187                self.db.commit()
188                print(f"Nombre actualizado: {nuevo_nombre}")
189        except Exception as e:
190            self.db.rollback()
191            self.mostrar_error(f"Error al actualizar nombre: {e}")

193    def recalcular_promedios_globales(self):
194        """
195        Recalcula las notas numéricas basándose en los nuevos pesos y luego
196        DELEGA la actualización de estados al MatriculaModel centralizado.
197        """
198        try:
199            # 1. Preparar datos de pesos
200            esquema = self.db.query(EvaluacionCurso).filter_by(curso_id=self.curso_id).all()
201            mapa_pesos = {e.id: e.porcentaje for e in esquema}
202
203            # 2. Recalcular SOLO la Nota Final Numérica
204            matriculas = self.db.query(Matricula).filter_by(curso_id=self.curso_id).all()
205
206            for mat in matriculas:
207                # Si abandonó, saltamos cálculo (la lógica de estados lo manejará después si es necesario)
208                if mat.estado == "NO REALIZO":
209                    continue
210
211                notas = self.db.query(Calificacion).filter_by(matricula_id=mat.id).all()
212
213                # Armar estructura para cálculo
214                lista_notas = []
215                for nota in notas:
216                    lista_notas.append({
217                        'puntaje': nota.puntaje,
218                        'peso': mapa_pesos.get(nota.evaluacion_curso_id, 0.0)
219                    })
220
221                # Calculamos el promedio numérico
222                nuevo_promedio = self.matricula_model.calcular_nota_ponderada(lista_notas)
223                mat.nota_final = nuevo_promedio
224
225            # 3. Guardamos los cambios de NOTAS en la BD
226            self.db.commit()
227
228            # 4. LLAMADA CENTRALIZADA PARA ESTADOS
229            self.matricula_model.actualizar_estados_por_curso(curso_id=self.curso_id)
230
231            print("Recálculo masivo (Notas + Estados Centralizados) completado.")
232
233        except Exception as e:
234            self.db.rollback()
235            print(f"Error en recálculo masivo: {e}")

237    def actualizar_porcentaje_bd(self, eval_id, nuevo_valor):
238        """Actualiza el peso porcentual en la BD y recalcula promedios globales."""
239        try:
240            ev = self.db.query(EvaluacionCurso).get(eval_id)
241            if ev:

243                self.db.commit()

247            print(f"Error al actualizar porcentaje: {e}")

249    def recalcular_total_ui(self):
250        """Suma los valores de los spinners de la tabla para actualizar el label de total."""
251        total = 0.0
252        for i in range(self.tabla.rowCount()):
253            widget = self.tabla.cellWidget(i, 1)
254            if isinstance(widget, QDoubleSpinBox):
255                total += widget.value()
256        self.actualizar_label_total(total)

258    def actualizar_label_total(self, total_pct):
259        """
260        Actualiza el texto y color del label de total de ponderación.
261        Habilita el botón de finalizar solo si suma 100%.
262        """
263        self.lbl_total.setText(f"Total Ponderación: {total_pct:.2f}%")
264        es_valido = abs(total_pct - 100.0) < 0.01
265
266        if es_valido:
267            self.lbl_total.setStyleSheet("color: green; font-weight: bold;")
268        elif total_pct > 100.0:
269            self.lbl_total.setStyleSheet("color: red; font-weight: bold;")
270        else:
271            self.lbl_total.setStyleSheet("color: orange; font-weight: bold;")
272
273        if hasattr(self, 'btn_cerrar'):
274            self.btn_cerrar.setEnabled(es_valido)

276    def agregar_evaluacion(self):
277        """Crea una nueva actividad evaluativa y la guarda en la BD."""
278        # Sanitizamos explícitamente antes de guardar, por si acaso
279        nombre = Sanitizer.limpiar_texto(self.input_nombre.text())
280        pct = self.input_porcentaje.value()
281
282        if not nombre:
283            self.mostrar_error("Debe ingresar un nombre para la actividad.")
284            return
285
286        try:
287            nueva = EvaluacionCurso(
288                curso_id=self.curso_id, nombre=nombre, porcentaje=pct,
289                orden=self.tabla.rowCount() + 1
290            )
291            self.db.add(nueva)
292            self.db.commit()
293
294            self.input_nombre.clear()
295            self.cargar_esquema()
296            self.recalcular_promedios_globales()
297        except Exception as e:
298            self.db.rollback()
299            self.mostrar_error(f"Error al guardar: {e}")

301    def eliminar_evaluacion(self, eval_id):
302        """Elimina una evaluación y sus notas asociadas tras confirmación."""
303        confirm = QMessageBox.question(self, "Confirmar", "¿Eliminar esta evaluación? Se borrarán las notas asociadas.",
304                                       QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No)
305        if confirm == QMessageBox.StandardButton.Yes:
306            try:
307                item = self.db.query(EvaluacionCurso).get(eval_id)
308                if item:
309                    self.db.delete(item)
310                    self.db.commit()
311                    self.cargar_esquema()
312                    self.recalcular_promedios_globales()
313            except Exception as e:
314                self.db.rollback()
315                self.mostrar_error(str(e))

def recalcular_promedios_globales(self): 
194    def recalcular_promedios_globales(self):
195        """
196        Recalcula las notas numéricas basándose en los nuevos pesos y luego
197        DELEGA la actualización de estados al MatriculaModel centralizado.
198        """
199        try:
200            # 1. Preparar datos de pesos
201            esquema = self.db.query(EvaluacionCurso).filter_by(curso_id=self.curso_id).all()
202            mapa_pesos = {e.id: e.porcentaje for e in esquema}
203
204            # 2. Recalcular SOLO la Nota Final Numérica
205            matriculas = self.db.query(Matricula).filter_by(curso_id=self.curso_id).all()
206
207            for mat in matriculas:
208                # Si abandonó, saltamos cálculo (la lógica de estados lo manejará después si es necesario)
209                if mat.estado == "NO REALIZO":
210                    continue
211
212                notas = self.db.query(Calificacion).filter_by(matricula_id=mat.id).all()
213
214                # Armar estructura para cálculo
215                lista_notas = []
216                for nota in notas:
217                    lista_notas.append({
218                        'puntaje': nota.puntaje,
219                        'peso': mapa_pesos.get(nota.evaluacion_curso_id, 0.0)
220                    })
221
222                # Calculamos el promedio numérico
223                nuevo_promedio = self.matricula_model.calcular_nota_ponderada(lista_notas)
224                mat.nota_final = nuevo_promedio
225
226            # 3. Guardamos los cambios de NOTAS en la BD
227            self.db.commit()
228
229            # 4. LLAMADA CENTRALIZADA PARA ESTADOS
230            self.matricula_model.actualizar_estados_por_curso(curso_id=self.curso_id)
231
232            print("Recálculo masivo (Notas + Estados Centralizados) completado.")
233
234        except Exception as e:
235            self.db.rollback()
236            print(f"Error en recálculo masivo: {e}")
