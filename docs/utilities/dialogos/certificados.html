315    def _formatear_fecha_mes_anio(self, fecha):
316        """
317        Formatea una fecha como 'MES DE AÑO' (Ej: ENERO DE 2026).
318        """
319        meses = ["ENERO", "FEBRERO", "MARZO", "ABRIL", "MAYO", "JUNIO",
320                 "JULIO", "AGOSTO", "SEPTIEMBRE", "OCTUBRE", "NOVIEMBRE", "DICIEMBRE"]
321        y, m = 0, 0
322        if isinstance(fecha, QDate):
323            y, m = fecha.year(), fecha.month()
324        elif hasattr(fecha, 'year'):
325            y, m = fecha.year, fecha.month
326        else:
327            return str(fecha)
328        return f"{meses[m - 1]} DE {y}"

340    def _formatear_fecha_larga(self, fecha):
341        """
342        Genera formato largo de fecha: '01 de enero de 2026'.
343        """
344        meses = ["enero", "febrero", "marzo", "abril", "mayo", "junio",
345                 "julio", "agosto", "septiembre", "octubre", "noviembre", "diciembre"]
346        d, m, y = 0, 0, 0
347        if isinstance(fecha, QDate):
348            d, m, y = fecha.day(), fecha.month(), fecha.year()
349        elif hasattr(fecha, 'year'):
350            d, m, y = fecha.day, fecha.month, fecha.year
351        else:
352            return str(fecha)
353
354        # d:02d asegura que salga '01' en lugar de '1'. Usamos 'de' en minúsculas.
355        return f"{d:02d} de {meses[m - 1]} de {y}"

357    def _generar_siglas_curso(self, nombre_curso):
358        """
359        Genera siglas de 3 letras o más para el curso, ignorando palabras comunes.
360        Ej: "CURSO DE PRIMEROS AUXILIOS" -> "PAU" (Dependiendo de la lógica exacta).
361        """
362        ignore = {'DE', 'DEL', 'LA', 'EL', 'EN', 'Y', 'PARA', 'CON', 'LOS', 'LAS', 'POR', 'TALLER', 'CURSO'}
363        clean_name = Sanitizer.limpiar_texto(nombre_curso).upper()
364        words = [w for w in clean_name.split() if w not in ignore]
365        if not words: return "CUR"
366        if len(words) >= 3:
367            return "".join(w[0] for w in words[:3])
368        elif len(words) == 2:
369            return (words[0][:1] + words[1][:2])
370        else:
371            w = words[0]
372            return w[:3] if len(w) >= 3 else w.ljust(3, 'X')

488            self.mostrar_error(f"Error en vista previa: {e}")
489        finally:
490            # Eliminamos la plantilla temporal, pase lo que pase
491            if os.path.exists(temp_tpl_dir):
492                shutil.rmtree(temp_tpl_dir)

358    def _generar_siglas_curso(self, nombre_curso):
359        """
360        Genera siglas de 3 letras o más para el curso, ignorando palabras comunes.
361        Ej: "CURSO DE PRIMEROS AUXILIOS" -> "PAU" (Dependiendo de la lógica exacta).
362        """
363        ignore = {'DE', 'DEL', 'LA', 'EL', 'EN', 'Y', 'PARA', 'CON', 'LOS', 'LAS', 'POR', 'TALLER', 'CURSO'}
364        clean_name = Sanitizer.limpiar_texto(nombre_curso).upper()
365        words = [w for w in clean_name.split() if w not in ignore]
366        if not words: return "CUR"
367        if len(words) >= 3:
368            return "".join(w[0] for w in words[:3])
369        elif len(words) == 2:
370            return (words[0][:1] + words[1][:2])
371        else:
372            w = words[0]
373            return w[:3] if len(w) >= 3 else w.ljust(3, 'X')
